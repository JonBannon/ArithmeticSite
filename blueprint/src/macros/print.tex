% This file starts with dummy macros that ensure the pdf

% macros.

% Make sure that arguments of \uses and \proves are real labels, by using invisible refs:
% latex prints a warning if the label is not defined, but nothing is shown in the pdf file.
% It uses LaTeX3 programming, this is why we use the expl3 package.
\ExplSyntaxOn
\NewDocumentCommand{\uses}{m}
 {\clist_map_inline:nn{#1}{\vphantom{\ref{##1}}}%
  \ignorespaces}
\NewDocumentCommand{\proves}{m}
 {\clist_map_inline:nn{#1}{\vphantom{\ref{##1}}}%
  \ignorespaces}
\ExplSyntaxOff

% Redefine theorem environments to handle blueprint [label=..., uses=...] options.
% Uses expl3 regex to extract only the label= value; ignores uses/proves and any
% continuation values after commas (e.g. uses=foo, bar, baz).
\ExplSyntaxOn

\cs_new:Nn \__bpthm_apply:n {
  % Labels are created by explicit \label{} in the theorem body (web-compatible).
  % The optional argument is accepted and discarded here for the print version.
}

\let \__bpthm_orig_theorem:        \theorem        \let \__bpthm_orig_endtheorem:        \endtheorem
\let \__bpthm_orig_proposition:    \proposition    \let \__bpthm_orig_endproposition:    \endproposition
\let \__bpthm_orig_lemma:          \lemma          \let \__bpthm_orig_endlemma:          \endlemma
\let \__bpthm_orig_corollary:      \corollary      \let \__bpthm_orig_endcorollary:      \endcorollary
\let \__bpthm_orig_definition:     \definition     \let \__bpthm_orig_enddefinition:     \enddefinition
\let \__bpthm_orig_remark:         \remark         \let \__bpthm_orig_endremark:         \endremark

\RenewDocumentEnvironment {theorem}     {O{}} {\__bpthm_orig_theorem:     \__bpthm_apply:n {#1}} {\__bpthm_orig_endtheorem:}
\RenewDocumentEnvironment {proposition} {O{}} {\__bpthm_orig_proposition: \__bpthm_apply:n {#1}} {\__bpthm_orig_endproposition:}
\RenewDocumentEnvironment {lemma}       {O{}} {\__bpthm_orig_lemma:       \__bpthm_apply:n {#1}} {\__bpthm_orig_endlemma:}
\RenewDocumentEnvironment {corollary}   {O{}} {\__bpthm_orig_corollary:   \__bpthm_apply:n {#1}} {\__bpthm_orig_endcorollary:}
\RenewDocumentEnvironment {definition}  {O{}} {\__bpthm_orig_definition:  \__bpthm_apply:n {#1}} {\__bpthm_orig_enddefinition:}
\RenewDocumentEnvironment {remark}      {O{}} {\__bpthm_orig_remark:      \__bpthm_apply:n {#1}} {\__bpthm_orig_endremark:}

\ExplSyntaxOff
