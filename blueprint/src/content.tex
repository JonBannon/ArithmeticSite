\chapter{The multiplicative monoid \texorpdfstring{$\Nplus$}{N+}}

The Arithmetic Site, introduced by Connes and Consani, is a geometric
object — a topos equipped with a structure sheaf — whose points in the
topos-theoretic sense are in natural bijection with the adèle class space
of $\mathbb{Q}$.  Its construction begins with a single, elementary
monoid: the positive integers under multiplication.  Everything else —
the tropical semiring, the presheaf topos, the structure sheaf, and
ultimately the connection to the Riemann zeta function — is built on top
of this one piece of data.

\begin{definition}[label=def:Nplus]
  Let $\Nplus$ denote the set $\{1, 2, 3, \ldots\}$ of strictly positive
  integers equipped with ordinary multiplication. This is a commutative
  monoid with unit $1$.
  In Lean we represent this as \texttt{PNat} with its multiplicative
  structure, via the abbreviation \texttt{NPos}.
\end{definition}

The unique factorisation theorem says that $\Nplus$ is the free
commutative monoid on the set of primes: every element factors uniquely
as a finite product $\prod_p p^{a_p}$.  This factorisation structure is
not merely background --- it will resurface as the combinatorial backbone
of the site structure on $B\Nplus$, where factorizations of integers into
coprime parts give rise to covering families whose overlap conditions
reflect divisibility.

\begin{definition}[label=def:BNplus, uses=def:Nplus]
  The \emph{classifying category} $B\Nplus$ of $\Nplus$ is the
  one-object category whose single object is $*$ and whose morphisms
  $* \to *$ are the elements of $\Nplus$, with composition given by
  multiplication.
\end{definition}

\begin{remark}
  The passage from a monoid $M$ to the one-object category $BM$ is the
  standard \emph{delooping} construction.  A presheaf on $BM$ — that is,
  a functor $BM^{\mathrm{op}} \to \Cat{Set}$ — is exactly a set $X$
  equipped with a left action of $M$: each element $n \in M$ acts as a
  function $X \to X$, and the action is compatible with composition.
  (Left and right actions coincide here since $\Nplus$ is commutative.)
  The presheaf topos $\PSh{\Nplus}$ introduced in
  Chapter~\ref{chap:presheaf_topos} is precisely the category of such
  sets-with-$\Nplus$-action, and it is in this topos that the Arithmetic
  Site lives.
\end{remark}

\chapter{The tropical semiring \texorpdfstring{$\Nbar$}{Nbar}}

The tropical semiring $\Nbar$ is the second fundamental ingredient of
the Arithmetic Site.  It plays the role of the \emph{structure sheaf}:
as Connes and Consani put it, $\Nbar$ is the semiring on which $\Nplus$
acts by multiplication, and it is this action that makes $\Nbar$ into
a sheaf of semirings over the topos $\widehat{B\Nplus}$.  The arithmetic
of $\Nbar$ is characteristic-$1$ arithmetic --- addition is idempotent
--- which is the tropical, or min-plus, shadow of ordinary arithmetic.
Connes and Consani define $\Nbar$ explicitly as $(\mathbb{N} \cup \{\infty\},
\inf, +)$, confirming the min-plus convention: addition is $\min$, with
$+\infty$ as the additive identity.  When elements of $\Nbar$ are written
as exponents $q^x$ for $q \in (0,1)$, the addition becomes $\max$ on
values (since smaller exponent means larger value), which is why the
stalks in the paper appear in max-plus form as $[0,1]_{\max}$ and
$\mathbb{R}_{\max}$.

\begin{definition}[label=def:Nbar]
  The \emph{tropical semiring} $\Nbar$ is the set
  $\mathbb{N} \cup \{+\infty\}$ equipped with:
  \begin{itemize}
    \item \emph{Tropical addition}: $a \oplus b := \min(a, b)$, with
          identity element $+\infty$.
    \item \emph{Tropical multiplication}: $a \odot b := a + b$
          (ordinary addition), with identity element $0$.
  \end{itemize}
  In Lean, $\Nbar$ is represented as \texttt{Tropical (WithTop $\mathbb{N}$)},
  via the abbreviation \texttt{NBar}.
  The zero element is \texttt{trop $\top$} (i.e.\ $+\infty$) and the
  multiplicative identity is \texttt{trop 0}.
\end{definition}

\begin{lemma}[label=lem:Nbar_semiring, uses=def:Nbar]
  $\Nbar$ is a commutative semiring.
\end{lemma}
\begin{proof}
  This is the Mathlib instance
  \texttt{CommSemiring (Tropical (WithTop $\mathbb{N}$))},
  found by \texttt{inferInstance}.
\end{proof}

\begin{lemma}[label=lem:Nbar_idempotent, uses=def:Nbar]
  Tropical addition on $\Nbar$ is idempotent: $a \oplus a = a$ for all
  $a \in \Nbar$.
\end{lemma}
\begin{proof}
  $a \oplus a = \min(a,a) = a$. In Lean this follows from
  \texttt{Tropical.add\_eq\_left}.
\end{proof}

\begin{remark}
  Idempotency of addition --- equivalently, the identity $1 \oplus 1 = 1$
  --- is the defining feature of \emph{characteristic $1$} arithmetic.
  It is what places $\Nbar$ in the same conceptual family as finite fields
  $\mathbb{F}_p$ (characteristic $p$), with $\min$ playing the role that
  $+\bmod p$ plays there.  The scaling endomorphisms defined below are the
  characteristic-$1$ analogue of the Frobenius endomorphism $x \mapsto x^p$
  of $\mathbb{F}_p$.
\end{remark}

\begin{definition}[label=def:scaling_action, uses=def:Nbar]
  For each $n \in \Nplus$, the \emph{scaling endomorphism}
  $\varphi_n : \Nbar \to \Nbar$ is defined by
  \[
    \varphi_n(x) := nx \quad (x \in \mathbb{N}),
    \qquad \varphi_n(+\infty) := +\infty,
  \]
  where $nx$ denotes ordinary multiplication in $\mathbb{N}$.
  Equivalently, $\varphi_n(x) = x^{\odot n}$, the $n$-fold tropical
  product of $x$ with itself.
  In Lean, $\varphi_n$ is defined as a semiring endomorphism
  \texttt{NBar~->+*~NBar} via \texttt{scalingEndomorphism}.
\end{definition}

\begin{lemma}[label=lem:scaling_semiring_hom, uses=def:scaling_action]
  For each $n \in \Nplus$, the map $\varphi_n$ is a semiring endomorphism
  of $\Nbar$: it preserves both operations and both identities.
\end{lemma}
\begin{proof}
  Preservation of $+\infty$ and of tropical multiplication are immediate.
  Preservation of tropical addition uses $n \cdot \min(a,b) = \min(na,nb)$
  for $n > 0$, which holds because scaling by $n$ is order-preserving on
  $\mathbb{N} \cup \{+\infty\}$.  In Lean this is the \texttt{map\_add'}
  field of \texttt{scalingEndomorphism}, proved via monotonicity of
  \texttt{nsmul} and a case split on \texttt{le\_total}.
\end{proof}

\begin{lemma}[label=lem:scaling_comp, uses=def:scaling_action]
  The scaling endomorphisms compose multiplicatively:
  \[
    \varphi_m \circ \varphi_n = \varphi_{mn}
  \]
  for all $m, n \in \Nplus$.  In particular, the map $n \mapsto \varphi_n$
  is a monoid homomorphism from $(\Nplus, \times)$ into the semiring
  endomorphism monoid $\mathrm{End}_{\mathrm{Semiring}}(\Nbar)$.
\end{lemma}
\begin{proof}
  $(\varphi_m \circ \varphi_n)(x) = m(nx) = (mn)x = \varphi_{mn}(x)$.
  In Lean this is \texttt{scalingEndomorphism\_comp}, proved via
  \texttt{pow\_mul}.
\end{proof}

\begin{remark}
  The Frobenius \emph{correspondences} $\Psi(\lambda)$, parametrised by
  $\lambda \in \Qplus$, are a more refined object: they live in the
  \emph{square} $\widehat{B{\Nplus}^2}$ of the Arithmetic Site and are
  defined via the action of $\Nplus \times \Nplus$ on
  $\Nbar \otimes_{\mathbb{B}} \Nbar$.  They are introduced in
  Chapter~\ref{chap:frobenius_correspondences}.
\end{remark}

\chapter{The presheaf topos \texorpdfstring{$\PSh{\Nplus}$}{PSh(N+)}}
\label{chap:presheaf_topos}

\begin{definition}[label=def:presheaf_topos, uses=def:BNplus]
  The \emph{presheaf topos} $\PSh{\Nplus}$ is the category of functors
  $B{\Nplus}^{\mathrm{op}} \to \Cat{Set}$, i.e.\ the category of sets
  equipped with a left $\Nplus$-action. In Lean this is
  \texttt{Action (Type*) (Multiplicative $\mathbb{N}$+)}.
\end{definition}

\begin{lemma}[label=lem:presheaf_topos_is_topos, uses=def:presheaf_topos]
  $\PSh{\Nplus}$ is a Grothendieck topos.
\end{lemma}
\begin{proof}
  Any presheaf category on a small category is a Grothendieck topos.
\end{proof}

\chapter{The structure sheaf}

\begin{definition}[label=def:structure_sheaf,
                  uses=def:presheaf_topos, def:Nbar, def:scaling_action]
  The \emph{structure sheaf} $\mathcal{O}$ of the Arithmetic Site is the
  object of $\PSh{\Nplus}$ whose underlying set is $\Nbar$ and whose
  $\Nplus$-action is given by the scaling endomorphisms:
  $n \cdot x := \varphi_n(x) = nx$.
\end{definition}

\begin{lemma}[label=lem:structure_sheaf_action,
             uses=def:structure_sheaf, lem:scaling_comp]
  The assignment $n \mapsto \varphi_n$ defines a monoid homomorphism
  $(\Nplus, \times) \to \mathrm{End}_{\mathrm{Semiring}}(\Nbar)$,
  making $\Nbar$ into a semiring object of $\PSh{\Nplus}$ with
  $\Nplus$ acting by multiplication.
\end{lemma}

\chapter{The Arithmetic Site}

\begin{definition}[label=def:arithmetic_site,
                  uses=def:presheaf_topos, def:structure_sheaf]
  The \emph{Arithmetic Site} is the pair $(\PSh{\Nplus}, \mathcal{O})$
  consisting of the presheaf topos $\PSh{\Nplus}$ together with its
  structure sheaf $\mathcal{O}$.
\end{definition}
