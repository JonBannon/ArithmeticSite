\chapter{The multiplicative monoid \texorpdfstring{$\Nplus$}{N+}}

\begin{definition}[label=def:Nplus]
  Let $\Nplus$ denote the set $\{1, 2, 3, \ldots\}$ of strictly positive
  integers equipped with ordinary multiplication. This is a commutative
  monoid with unit $1$.
  In Lean we represent this as \texttt{PNat} with its multiplicative
  structure, via the abbreviation \texttt{NPos}.
\end{definition}

\begin{definition}[label=def:BNplus, uses=def:Nplus]
  The \emph{classifying category} $B\Nplus$ of $\Nplus$ is the
  one-object category whose single object is $*$ and whose morphisms
  $* \to *$ are the elements of $\Nplus$, with composition given by
  multiplication.
\end{definition}

\chapter{The tropical semiring \texorpdfstring{$\Nbar$}{Nbar}}

\begin{definition}[label=def:Nbar]
  The \emph{tropical semiring} $\Nbar$ is the set
  $\mathbb{N} \cup \{+\infty\}$ equipped with:
  \begin{itemize}
    \item \emph{Tropical addition}: $a \oplus b := \min(a, b)$, with
          identity element $+\infty$.
    \item \emph{Tropical multiplication}: $a \odot b := a + b$
          (ordinary addition), with identity element $0$.
  \end{itemize}
  In Lean, $\Nbar$ is represented as \texttt{Tropical (WithTop ℕ)},
  via the abbreviation \texttt{NBar}.
  The zero element is \texttt{trop ⊤} (i.e.\ $+\infty$) and the
  multiplicative identity is \texttt{trop 0}.
\end{definition}

\begin{lemma}[label=lem:Nbar_semiring, uses=def:Nbar]
  $\Nbar$ is a commutative semiring.
\end{lemma}
\begin{proof}
  This is the Mathlib instance \texttt{CommSemiring (Tropical (WithTop ℕ))},
  found by \texttt{inferInstance}.
\end{proof}

\begin{lemma}[label=lem:Nbar_idempotent, uses=def:Nbar]
  Tropical addition on $\Nbar$ is idempotent: $a \oplus a = a$ for all
  $a \in \Nbar$.
\end{lemma}
\begin{proof}
  $a \oplus a = \min(a,a) = a$. In Lean this follows from
  \texttt{Tropical.add\_eq\_left}.
\end{proof}

\begin{definition}[label=def:frobenius, uses=def:Nbar]
  For each $n \in \Nplus$, the \emph{Frobenius endomorphism}
  $\varphi_n : \Nbar \to \Nbar$ is defined by
  \[
    \varphi_n(x) := n + x \quad (x \in \mathbb{N}),
    \qquad \varphi_n(+\infty) := +\infty,
  \]
  where $+$ denotes ordinary addition in $\mathbb{N} \cup \{+\infty\}$.
  Equivalently, in \texttt{Tropical (WithTop ℕ)}, this is tropical
  multiplication by \texttt{trop (n : WithTop ℕ)}, i.e.\
  $\varphi_n(x) = \texttt{trop}\, n \odot x$.

  In Lean, $\varphi_n$ is defined as an \texttt{AddMonoid.End NBar}
  via \texttt{frobeniusEndomorphism}.
\end{definition}

\begin{lemma}[label=lem:frobenius_addmon_hom, uses=def:frobenius]
  For each $n \in \Nplus$, the map $\varphi_n : \Nbar \to \Nbar$ is an
  additive monoid endomorphism, i.e.\ it preserves tropical addition and
  the zero element $+\infty$:
  \begin{align*}
    \varphi_n(+\infty) &= +\infty, \\
    \varphi_n(a \oplus b) &= \varphi_n(a) \oplus \varphi_n(b).
  \end{align*}
  The second identity follows from the distributivity
  $n + \min(a,b) = \min(n+a, n+b)$.
\end{lemma}

\begin{remark}
  Note that $\varphi_n$ is \emph{not} a semiring homomorphism in general,
  since it does not preserve the multiplicative identity:
  $\varphi_n(0) = n \neq 0$ for $n > 0$.
  It is however a homomorphism of $\Nbar$-modules.
\end{remark}

\begin{lemma}[label=lem:frobenius_comp, uses=def:frobenius]
  The Frobenius endomorphisms satisfy the composition law
  \[
    \varphi_m \circ \varphi_n = \varphi_{m+n}
  \]
  for all $m, n \in \Nplus$. In particular, the map
  $n \mapsto \varphi_n$ is a semigroup homomorphism from
  $(\Nplus, +)$ into the endomorphism semigroup
  $\mathrm{End}_{\mathrm{AddMon}}(\Nbar)$.
\end{lemma}
\begin{proof}
  For any $x \in \Nbar$:
  $(\varphi_m \circ \varphi_n)(x) = m + (n + x) = (m+n) + x = \varphi_{m+n}(x)$.
  In Lean this is \texttt{frobenius\_comp}, proved via \texttt{mul\_assoc}
  in \texttt{Tropical (WithTop ℕ)}.
\end{proof}

\chapter{The presheaf topos \texorpdfstring{$\PSh{\Nplus}$}{PSh(N+)}}

\begin{definition}[label=def:presheaf_topos, uses=def:BNplus]
  The \emph{presheaf topos} $\PSh{\Nplus}$ is the category of functors
  $B\Nplus^{\mathrm{op}} \to \Cat{Set}$, i.e.\ the category of sets
  equipped with a left $\Nplus$-action. In Lean this is
  \texttt{Action (Type*) (Multiplicative ℕ+)}.
\end{definition}

\begin{lemma}[label=lem:presheaf_topos_is_topos, uses=def:presheaf_topos]
  $\PSh{\Nplus}$ is a Grothendieck topos.
\end{lemma>
\begin{proof}
  Any presheaf category on a small category is a Grothendieck topos.
\end{proof}

\chapter{The structure sheaf}

\begin{definition}[label=def:structure_sheaf,
                  uses=def:presheaf_topos, def:Nbar, def:frobenius]
  The \emph{structure sheaf} $\mathcal{O}$ of the Arithmetic Site is the
  object of $\PSh{\Nplus}$ whose underlying set is $\Nbar$ and whose
  $\Nplus$-action is given by the Frobenius endomorphisms:
  $n \cdot x := \varphi_n(x)$.
\end{definition}

\begin{lemma}[label=lem:structure_sheaf_action,
             uses=def:structure_sheaf, lem:frobenius_comp]
  The assignment $n \mapsto \varphi_n$ defines a semigroup homomorphism
  $(\Nplus, +) \to \mathrm{End}_{\mathrm{AddMon}}(\Nbar)$,
  making $\Nbar$ into a left $(\Nplus, +)$-module.
\end{lemma>

\chapter{The Arithmetic Site}

\begin{definition}[label=def:arithmetic_site,
                  uses=def:presheaf_topos, def:structure_sheaf]
  The \emph{Arithmetic Site} is the pair $(\PSh{\Nplus}, \mathcal{O})$
  consisting of the presheaf topos $\PSh{\Nplus}$ together with its
  structure sheaf $\mathcal{O}$.
\end{definition}
